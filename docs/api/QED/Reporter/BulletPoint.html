def error(step, exception)
  raise exception if $DEBUG
  msg = ''
  msg << "  ##### ERROR #####\n"
  msg << "  # " + exception.to_s + "\n"
  msg << "  # " + clean_backtrace(exception.backtrace[0])
  msg = msg.ansi(:magenta)
  io.puts msg
  io.print "#{step.example}".ansi(:red)
end
