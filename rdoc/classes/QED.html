  def self.root(path=nil)
    path ||= Dir.pwd
    path = File.dirname(path) unless File.directory?(path)
    until path == File.dirname(path)
      mark = Dir[File.join(path, 'README*')].first
      return path if mark
      path = File.dirname(path)
    end
    nil
  end
