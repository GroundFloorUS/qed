    def report_error(step, exception)
      raise exception if $DEBUG
      msg = ''
      msg << "  ##### ERROR #####\n"
      msg << "  # " + exception.to_s + "\n"
      msg << "  # " + exception.backtrace[0]
      msg = ANSICode.magenta(msg)
      io.puts msg
      #io.puts
      io.puts ANSICode.red("#{step}")
    end
