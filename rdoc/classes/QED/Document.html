    def file_type(text)
      rdoc = text.index(/^\=/)
      markdown = text.index(/^\#/)
      if markdown && rdoc
        rdoc < markdown ? '.rdoc' : '.markdown'
      elsif rdoc
        '.rdoc'
      elsif markdown
        '.markdown'
      else  # fallback to rdoc
        '.rdoc'
      end
    end
